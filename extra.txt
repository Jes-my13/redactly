%{
#include <stdio.h>
#include <string.h>

char keywords[10][10] = {"int","float","char","if","else","while","for","do","return","main"};
int i, flag;
%}
%%
[a-zA-Z_][a-zA-Z0-9_]* {
    flag = 0;
    for(i = 0; i < 10; i++) {
        if(strcmp(yytext, keywords[i]) == 0) {
            printf("%s is a KEYWORD\n", yytext);
            flag = 1;
            break;
        }
    }
    if(!flag)
        printf("%s is an IDENTIFIER\n", yytext);
}
[0-9]+ {
    printf("%s is a NUMBER\n", yytext);
}
[ \t\n]+    ;
.           ;
%%
int main() {
    printf("Enter a line of code:\n");
    yylex();
    return 0;
}

%{
#include <stdio.h>
#include <string.h>
char str[100];
%}
%%
[a-zA-Z]+ {
    strcpy(str, yytext);
    int i = 0, j = strlen(str) - 1;
    int flag = 1;
    while(i < j) {
        if(str[i] != str[j]) {
            flag = 0;
            break;
        }
        i++; j--;
    }
    if(flag)
        printf("%s is a Palindrome\n", str);
    else
        printf("%s is not a Palindrome\n", str);
}
\n { }
%%
int main() {
    printf("Enter a string:\n");
    yylex();
    return 0;
}



%{
#include <stdio.h>
int digits = 0, alphabets = 0, spaces = 0, special = 0;
%}
%%
[0-9]         { digits++; }
[a-zA-Z]      { alphabets++; }
[ \t\n]       { spaces++; }
.             { special++; }
%%
int main() {
    printf("Enter text:\n");
    yylex();
    printf("Digits           : %d\n", digits);
    printf("Alphabets        : %d\n", alphabets);
    printf("Spaces           : %d\n", spaces);
    printf("Special Characters : %d\n", special);
    return 0;
}
