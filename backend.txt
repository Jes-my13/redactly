#include <stdio.h>
#include <string.h>

int main()
{
    char icode[10][30], str[20], opr[10];
    int i = 0;

    printf("Intermediate code at exit:\n");

    // Read intermediate code until "exit" is encountered
    do {
        scanf("%s", icode[i]);
    } while (strcmp(icode[i++], "exit") != 0);

    printf("\nTarget code generation\n");

    i = 0;
    do {
        strcpy(str, icode[i]);

        // Choose operation based on the operator in the string
        switch (str[3]) {
            case '+':
                strcpy(opr, "ADD");
                break;
            case '-':
                strcpy(opr, "SUB");
                break;
            case '*':
                strcpy(opr, "MUL");
                break;
            case '/':
                strcpy(opr, "DIV");
                break;
            default:
                strcpy(opr, "???");
                break;
        }

        // Target code instructions
        printf("\n\tMov %c, R%d", str[2], i);
        printf("\n\t%s %c, R%d", opr, str[4], i);
        printf("\n\tMov R%d, %c", i, str[0]);

    } while (strcmp(icode[++i], "exit") != 0);

    return 0;
}
