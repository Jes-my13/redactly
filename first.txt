#include <stdio.h>
#include <string.h>
#include <ctype.h>

#define MAX 20

int n;
char production[MAX][MAX];
char firstSet[MAX][MAX];
int firstCount[MAX];

void addToFirst(int index, char symbol) {
    for (int i = 0; i < firstCount[index]; i++) {
        if (firstSet[index][i] == symbol)
            return;  // avoid duplicates
    }
    firstSet[index][firstCount[index]++] = symbol;
}

void findFirst(char nonTerminal, int index) {
    for (int i = 0; i < n; i++) {
        if (production[i][0] == nonTerminal) {
            // If production is A -> #
            if (production[i][2] == '#') {
                addToFirst(index, '#');
            } else {
                int j = 2;
                while (production[i][j] != '\0') {
                    char symbol = production[i][j];

                    if (!isupper(symbol)) {
                        // If symbol is terminal
                        addToFirst(index, symbol);
                        break;
                    } else {
                        // If symbol is non-terminal
                        int foundEpsilon = 0;
                        findFirst(symbol, symbol - 'A');

                        for (int k = 0; k < firstCount[symbol - 'A']; k++) {
                            if (firstSet[symbol - 'A'][k] != '#') {
                                addToFirst(index, firstSet[symbol - 'A'][k]);
                            } else {
                                foundEpsilon = 1;
                            }
                        }

                        if (!foundEpsilon)
                            break;

                        j++;
                        if (production[i][j] == '\0') {
                            addToFirst(index, '#');
                        }
                    }
                }
            }
        }
    }
}

int main() {
    printf("Enter number of productions: ");
    scanf("%d", &n);

    printf("Enter productions (use # for epsilon):\n");
    for (int i = 0; i < n; i++) {
        scanf("%s", production[i]);
    }

    for (int i = 0; i < MAX; i++) {
        firstCount[i] = 0;
    }

    for (int i = 0; i < n; i++) {
        char nonTerminal = production[i][0];
        findFirst(nonTerminal, nonTerminal - 'A');
    }

    printf("\nFIRST sets:\n");
    int printed[26] = {0};

    for (int i = 0; i < n; i++) {
        char nonTerminal = production[i][0];
        if (!printed[nonTerminal - 'A']) {
            printf("FIRST(%c) = { ", nonTerminal);
            for (int j = 0; j < firstCount[nonTerminal - 'A']; j++) {
                printf("%c, ", firstSet[nonTerminal - 'A'][j]);
            }
            printf("\b\b }\n");  // remove last comma
            printed[nonTerminal - 'A'] = 1;
        }
    }

    return 0;
}
